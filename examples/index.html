animo

JavaScript Animation Framework

animo
.create("over", {
    css: "background-color: #A36060; width: 250px; box-shadow: 5px -5px 10px 1px #000;",
    duration: 300
})
.create("out", {
    css: "background-color: #999; width: 150px; box-shadow: none;",
    duration: 500,
    ease: "ease-out-back"
})

$(".demo-nav > li > a")
.mouseover(function() {
    animo.play(this, "over");
})
.mouseout(function() {
    animo.play(this, "out");
});
